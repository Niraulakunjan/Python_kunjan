Numeric Types

Integer (int)

# Integer examples
x = 10
y = -5
big_num = 1_000_000  # Using underscore for readability

print(type(x))       # Output: 
print(x + y)         # Output: 5
print(big_num)       # Output: 1000000

Float

# Float examples
pi = 3.14159
scientific = 2.5e-3  # Scientific notation

print(type(pi))      # Output: 
print(pi + 1)        # Output: 4.14159
print(scientific)     # Output: 0.0025

Complex

# Complex number examples
z1 = 2 + 3j
z2 = complex(1, -2)

print(type(z1))      # Output: 
print(z1 + z2)       # Output: (3+1j)
print(z1.real)       # Output: 2.0

Text Type (str)

# String examples
single_quotes = 'Hello'
double_quotes = "World"
multi_line = """This is a
multi-line string"""

# String operations
print(single_quotes + ' ' + double_quotes)  # Output: Hello World
print(len(multi_line))                      # Output: 24
print('Hello'[1:4])                         # Output: ell

Sequence Types

List

# List examples
fruits = ['apple', 'banana', 'orange']
numbers = [1, 2, 3, 4, 5]
mixed = [1, 'hello', 3.14, True]

# List operations
fruits.append('grape')              # Add item
print(fruits[0])                    # Output: apple
print(len(numbers))                 # Output: 5
print(mixed[1:3])                   # Output: ['hello', 3.14]

Tuple

# Tuple examples
coordinates = (10, 20)
rgb = (255, 128, 0)

# Tuple operations
print(coordinates[0])               # Output: 10
print(len(rgb))                     # Output: 3
print(coordinates + rgb)            # Output: (10, 20, 255, 128, 0)

Mapping Type (dict)

# Dictionary examples
person = {
    'name': 'John',
    'age': 30,
    'city': 'New York'
}

# Dictionary operations
print(person['name'])               # Output: John
person['email'] = 'john@email.com'  # Add new key-value pair
print(person.keys())                # Output: dict_keys(['name', 'age', 'city', 'email'])
print(person.get('phone', 'N/A'))   # Output: N/A (default value)

Set Types

# Set examples
fruits_set = {'apple', 'banana', 'orange'}
numbers_set = {1, 2, 3, 3, 4, 4, 5}  # Duplicates are removed

# Set operations
fruits_set.add('grape')             # Add item
print(numbers_set)                   # Output: {1, 2, 3, 4, 5}
print(len(fruits_set))               # Output: 4

# Frozen set (immutable)
frozen = frozenset([1, 2, 3])

Boolean Type

# Boolean examples
is_active = True
is_logged_in = False

# Boolean operations
print(is_active and is_logged_in)    # Output: False
print(is_active or is_logged_in)     # Output: True
print(not is_active)                 # Output: False

# Boolean conversion
print(bool(1))       # Output: True
print(bool(""))      # Output: False
print(bool([]))      # Output: False

Type Conversion

# Type conversion examples
# String to number
num_str = "123"
num_int = int(num_str)      # Convert to integer
num_float = float(num_str)  # Convert to float

# Number to string
back_to_str = str(num_int)  # Convert back to string

# List/String/Tuple conversion
my_list = list("Hello")     # Convert string to list
my_tuple = tuple([1, 2, 3]) # Convert list to tuple
my_set = set([1, 2, 2, 3])  # Convert list to set

print(my_list)              # Output: ['H', 'e', 'l', 'l', 'o']
print(my_tuple)             # Output: (1, 2, 3)
print(my_set)               # Output: {1, 2, 3}
